随机一言 - API

# 请求方式 #
    Method: Get
# 请求地址 #
    {{ host_name }}api/yan
# 请求参数 #
    参数 内容 用途
    type 0 全部类型 | i 诗词 | y 网易云 | f 来自网络 一言类型 如果为空默认 0 全部类型
    charset gbk/utf-8 字符编码，默认utf-8
    encode text/js/json 返回方式 如果为空默认 json 返回
# 返回参数 #
    参数 内容
    id 一言ID
    yan 一言内容
    from_who 作者
    come_from 出处
    type 一言类型 i 诗词 | y 网易云 | f 来自网络
    length 一言长度
# 使用方法 #
    返回文字:    {{ host_name }}api/yan?charset=utf-8
    返回诗词:    {{ host_name }}api/yan?type=i
    js调用举例: <script src="{{ host_name }}api/yan?encode=js"></script>
    js输出举例: <div id="yan"><script>yan()</script></div>
